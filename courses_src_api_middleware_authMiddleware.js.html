const jwt = require("jsonwebtoken");

// Secret key used for JWT token signing and verification
const secretKey = "1234567890";

/**
 * Middleware for authenticating JWT tokens.
 * @namespace authMiddleware
 */

/**
 * Middleware function to verify JWT token.
 * @memberof authMiddleware
 * @param {*} req - The request object.
 * @param {*} res - The response object.
 * @param {*} next - The next middleware function.
 * @returns {JsonResponse} - Returns a JSON response.
 */
const authMiddleware = (req, res, next) => {
  try {
    // Extracting JWT token from Authorization header
    const token = req.headers.authorization.split(" ")[1];
    // Verifying and decoding JWT token
    const decoded = jwt.verify(token, secretKey);
    // Storing decoded user data in request object
    req.userData = decoded;
    next();
  } catch (error) {
    // Handling authentication failure
    return res.status(401).send({ message: "Authentication Failed" });
  }
};

module.exports = authMiddleware;
