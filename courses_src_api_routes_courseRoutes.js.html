const express = require("express");
const router = express.Router();
const courseController = require("../controllers/courseController");

/**
 * Router for managing courses.
 * @namespace courseRoutes
 */

/**
 * Route for creating a new course.
 * @memberof courseRoutes
 * @name POST /create
 * @function
 */
router.post("/create", courseController.createCourse);

/**
 * Route for editing an existing course.
 * @memberof courseRoutes
 * @name PUT /:id
 * @function
 */
router.put("/:id", courseController.editCourse);

/**
 * Route for deleting a course.
 * @memberof courseRoutes
 * @name DELETE /:id
 * @function
 */
router.delete("/:id", courseController.deleteCourse);

/**
 * Route for fetching all courses.
 * @memberof courseRoutes
 * @name GET /
 * @function
 */
router.get("/", courseController.getAllCourses);

/**
 * Route for fetching details of a specific course.
 * @memberof courseRoutes
 * @name GET /:id
 * @function
 */
router.get("/:id", courseController.getCourseDetails);

module.exports = router;
